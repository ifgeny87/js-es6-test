Что нового в ES6, отличия от ES5

  • var против let / const
  • Замена немедленно вызываемых функций (IIFE) на блоки (Blocks) 
    // ES5
(function() { ... })();
// ES6
{ ... }
  • Стрелочные функции
  • .include() и .repeat() для строк 
    '123'.includes(2); // true
'123'.repeat(3); // '123123123'
  • Шаблонные литералы с интерполяцией 
    let name = 'Vasya';
alert(`Hello, ${name}!`);
  • Деструктуризация массивов и объектов 
    let [a, b] = [1, 2, 3, 4]; // a = 1, b = 2
let {x, y} = {x: 100}; // x = 100, y = undefined
  • Модули - export / import 
    export let name = 'Jack';
export function hello(name) {}
// OR
export default {
	x: 100, 
	name: 'Vasya', 
	func: () => {}
};

import 'math';
import {x, name as text} from 'module';
import * as util from 'utils';
import util from 'utils';
import foo, {foo1, foo2} from 'foos';
  • Параметры по умолчанию
  • Rest parameter / Остаточные параметры 
    function foo(a, ...args) {
	for(let arg of args) { ... }
}
  • Именованные параметры 
    function f1({x=100, y}) {
	console.log(x, y)
}

function f2({x=100, y} = {}) {
	console.log(x, y)
}

f1(1); // 100 undefined
f1(); // Uncaught TypeError
f1({x: 1, y: 2, z: 3}); // 1 2
f2(1); // 100 undefined
f2(); // 100 undefined
f2({x: 1, y: 2, z: 3}); // 1 2
  • Операторы расширения // Math.max(...[1, 2, 3])
  • Классы
  • Мэпы 
    let map = new Map([ ['name', 'David'], [true, 'TRUE'], [{}, 'object'] ]);
map.has('name'); // true
map.set('name', 'Vasiliy');
map.get('name'); // Vasiliy

for(let [key, value] of map.entries()) {
	console.log(key, value);
}
  • Promises 
    function foo(a) {
	return new Promise((ok, rej) => {
    	isFinite(a) && ok(a + 100);
		rej('A must be number');
	})
}

Promise.race([foo(1), foo(2), foo(3)])
	.then(console.info)
	.catch(console.error)
	.finally(x => console.log('Ok', x))
  • Генераторы 
    function* gen(a=98) {
	let input;
	while(a < 100) {
		input = yield a++;
	}
	yield input;
}

let g = gen();
console.log(g.next(2)); // {value: 98, done: false}
console.log(g.next(3)); // {value: 99, done: false}
console.log(g.next(4)); // {value: 4, done: false}
console.log(g.next(5)); // {value: undefined, done: true}
  • Async / Await 
    async function f(a) {
	await console.log(a+1)
	await console.log(a+2)
	console.log(a+3)
}
f(10)
f(100)
f(1000)
console.log('done')
// Result: 11, 101, 1001, done
// Then: 12, 102, 1002
// Then: 13, 103, 1003
  • Getters / Setters 
    // Classes
class Person {
	get name() { ... }
	set name(newName) { ... }
}
// Objects
let p = {
	_name: 'Jack',
	get name() { ... },
	set name(new Name) { ... }
}
  • Symbol / Symbol.for 
    Symbol('a') == Symbol('a') // false
Symbol('a') == Symbol.for('a') // false
Symbol.for('a') == Symbol.for('a') // true
